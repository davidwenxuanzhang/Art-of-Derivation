# Integral with $ax^2+b$

### 22
$$
\int \frac{1}{ax^2+b} \ \mathrm{d}x = 
\left\{\begin{aligned}
&\frac{1}{\sqrt{ab}}\arctan\sqrt{\frac{a}{b}}x+C \quad (b>0)\\
&\frac{1}{2\sqrt{-ab}}\ln\left|\frac{\sqrt{a}x-\sqrt{-b}}{\sqrt{a}x+\sqrt{-b}}\right|+C \quad (b<0)
\end{aligned}
\right.
$$
<br/>
<details>
<summary><span className="math-keyword">Derivation</span></summary>

$$
I=\int \frac{1}{ax^2+b} \ \mathrm{d}x =\int{\frac{1}{(\sqrt{a}x)^2+b}}\ \mathrm{d}x
$$
When $b>0$ :
$$
\begin{align*}
    \int{\frac{1}{(\sqrt{a}x)^2+b}}\ \mathrm{d}x&=\int{\frac{1}{(\sqrt{a}x)^2+(\sqrt{b})^2}}\ \mathrm{d}x\\
                                                &=\int{\frac{1}{(\sqrt{a}x)^2+(\sqrt{b})^2}}\ \frac{\mathrm{d}(\sqrt{a}x)}{\sqrt{a}} &&\text{(Direct Substitution)}\\
                                                &=\frac{1}{\sqrt{a}}\int{\frac{1}{(\sqrt{a}x)^2+(\sqrt{b})^2}}\ \mathrm{d}x\\
                                                &=\frac{1}{\sqrt{ab}}\arctan\sqrt{\frac{a}{b}}x+C &&(\text{Integral Formula \#19})
\end{align*}
$$
When $b<0$ :
$$
\begin{align*}
    \int{\frac{1}{(\sqrt{a}x)^2+b}}\ \mathrm{d}x&=\int{\frac{1}{(\sqrt{a}x)^2-(\sqrt{-b})^2}}\ \mathrm{d}x\\
                                                &=\int{\frac{1}{(\sqrt{a}x)^2-(\sqrt{-b})^2}}\ \frac{\mathrm{d}(\sqrt{a}x)}{\sqrt{a}} &&\text{(Direct Substitution)}\\
                                                &=\frac{1}{\sqrt{a}}\int{\frac{1}{(\sqrt{a}x)^2-(\sqrt{-b})^2}}\ \mathrm{d}x\\
                                                &\frac{1}{2\sqrt{-ab}}\ln\left|\frac{\sqrt{a}x-\sqrt{-b}}{\sqrt{a}x+\sqrt{-b}}\right|+C &&(\text{Integral Formula \#21})
\end{align*}
$$
Hence: 
$$
\int \frac{1}{ax^2+b} \ \mathrm{d}x = 
\left\{\begin{aligned}
&\frac{1}{\sqrt{ab}}\arctan\sqrt{\frac{a}{b}}x+C \quad (b>0)\\
&\frac{1}{2\sqrt{-ab}}\ln\left|\frac{\sqrt{a}x-\sqrt{-b}}{\sqrt{a}x+\sqrt{-b}}\right|+C \quad (b<0)
\end{aligned}
\right.
\tag*{$\blacksquare$}
$$

### 23
$$
\int\frac{x}{ax^2+b}\ \mathrm{d}x=\frac{1}{2a}\ln\left|ax^2+b\right|+C
$$
<br/>
<details>
<summary><span className="math-keyword">Derivation</span></summary>

$$
\begin{align*}
    I&=\int{\frac{x}{ax^2+b}}\ \mathrm{d}x\\
     &=\int{\frac{x}{ax^2+b}}\ \frac{\mathrm{d}(ax^2+b)}{2ax} &&\text{(Direct Substitution)}\\
     &=\frac{1}{2a}\int{\frac{1}{ax^2+b}}\ \mathrm{d}(ax^2+b)\\
     &=\frac{1}{2a}\ln\left|ax^2+b\right|+C \tag*{$\blacksquare$}
\end{align*}
$$

### 24
$$
\int \frac{x^2}{ax^2+b}\ \mathrm{d}x=\frac{x}{a}-\frac{b}{a}\int \frac{1}{ax^2+b}\ \mathrm{d}x
$$
<br/>
<details>
<summary><span className="math-keyword">Derivation</span></summary>
$$
\begin{align*}
    I&=\int{\frac{x^2}{ax^2+b}}\ \mathrm{d}x\\
     &=\int{\frac{ax^2+b-b}{a(ax^2+b)}}\ \mathrm{d}x\\
     &=\frac{1}{a}\left[\int{\frac{ax^2+b}{ax^2+b}}\ \mathrm{d}x-\int{\frac{b}{ax^2+b}}\ \mathrm{d}x \right]\\
     &=\frac{x}{a}-\frac{b}{a}\int \frac{1}{ax^2+b}\ \mathrm{d}x \tag*{$\blacksquare$}
\end{align*}
$$

### 25
$$
\int \frac{1}{x\left(ax^2+b\right)}\ \mathrm{d}x=\frac{1}{2b}\ln \frac{x^2}{\left| ax^2+b\right|}+C
$$
<br/>
<details>
<summary><span className="math-keyword">Derivation</span></summary>

$$
\begin{align*}
    I&=\int {\frac{1}{x\left(ax^2+b\right)}}\ \mathrm{d}x\\
     &=\int{\frac{x}{x^2(ax^2+b)}}\ \mathrm{d}x\\
     &=\int{\frac{1}{x^2(ax^2+b)}}\ \mathrm{d}\left(\frac{x^2}{2}\right) &&(x\ \mathrm{d}x=\mathrm{d}\left(\frac{x^2}{2}\right))\\
     &=\frac{1}{2}\int{\frac{1}{x^2(ax^2+b)}}\ \mathrm{d}(x^2)\\ 
     &=\frac{1}{2}\int{\left(\frac{A}{x^2}+\frac{B}{ax^2+b} \right)}\ \mathrm{d}(x^2) &&\text{(Partial Fraction Decomposition)}\\
     &=\frac{1}{2}\int{\frac{A(ax^2+b)+Bx^2}{x^2(ax^2+b)}}\ \mathrm{d}(x^2)\\
     &=\frac{1}{2}\int{\frac{(Aa+B)x^2+Ab}{x^2(ax^2+b)}}\ \mathrm{d}(x^2)
\end{align*}
$$
Comparing coefficient gives: 
$$
(Aa+B)x^2+Ab\equiv 1 \implies
\left\{
\begin{align*}
&Ab=1\\
&Aa+B=0
\end{align*}
\right.
\implies
\left\{
\begin{align*}
&A=\frac{1}{b}\\
&B=-\frac{a}{b}
\end{align*}
\right.
$$

Plug in back: 
$$
\begin{align*}
    I&=\frac{1}{2}\int{\left(\frac{\frac{1}{b}}{x^2}+\frac{-\frac{a}{b}}{ax^2+b} \right)}\ \mathrm{d}(x^2) \\
     &=\frac{1}{2}\left[\frac{1}{b}\int{\frac{1}{x^2}}\ \mathrm{d}(x^2)-\frac{a}{b}\int{\frac{1}{ax^2+b}}\ \frac{\mathrm{d}(ax^2+b)}{a} \right]  &&\text{(Direct Substitution)}\\
     &=\frac{1}{2b}\left(\ln{\left|x^2 \right|}-\ln{\left|ax^2+b \right|}\right)+C\\ 
     &=\frac{1}{2b}\ln \frac{x^2}{\left| ax^2+b\right|}+C \tag*{$\blacksquare$}
\end{align*}
$$

### 26
$$
\int \frac{1}{x^2\left(ax^2+b\right)}\ \mathrm{d}x=-\frac{1}{bx}-\frac{a}{b}\int \frac{1}{ax^2+b}\ \mathrm{d}x
$$
<br/>
<details>
<summary><span className="math-keyword">Derivation</span></summary>

$$
\begin{align*}
    I&=\int{ \frac{1}{x^2\left(ax^2+b\right)}}\ \mathrm{d}x\\
     &=\int{\left(\frac{A}{x^2}+\frac{B}{ax^2+b} \right)}\ \mathrm{d}x &&\text{(Partial Fraction Decomposition)}\\
     &=\int{\frac{A(ax^2+b)+Bx^2}{x^2(ax^2+b)}}\ \mathrm{d}x\\
     &=\int{\frac{(Aa+B)x^2+Ab}{x^2(ax^2+b)}}\ \mathrm{d}x\\
\end{align*}
$$
Comparing coefficient gives: 
$$
(Aa+B)x^2+Ab\equiv 1 \implies
\left\{
\begin{align*}
&Ab=1\\
&Aa+B=0
\end{align*}
\right.
\implies
\left\{
\begin{align*}
&A=\frac{1}{b}\\
&B=-\frac{a}{b}
\end{align*}
\right.
$$
Plug in back: 
$$
\begin{align*}
    I&=\int{\left(\frac{\frac{1}{b}}{x^2}+\frac{-\frac{a}{b}}{ax^2+b} \right)}\ \mathrm{d}x\\
     &=\frac{1}{b}\int{\frac{1}{x^2}}\ \mathrm{d}x -\frac{a}{b} \int{\frac{1}{ax^2+b}}\ \mathrm{d}x &&\text{(Direct Substitution)}\\
     &=-\frac{1}{bx}-\frac{a}{b}\int \frac{1}{ax^2+b}\ \mathrm{d}x  \tag*{$\blacksquare$}
\end{align*}
$$

### 27
$$
\int \frac{1}{x^3\left(ax^2+b\right)}\ \mathrm{d}x=\frac{a}{2b^2}\ln{\frac{\left|ax^2+b\right|}{x^2}}-\frac{1}{2bx^2}+C
$$
<br/>
<details>
<summary><span className="math-keyword">Derivation</span></summary>

$$
\begin{align*}
 I&=\int{\frac{1}{x^3(ax^2+b)}}\ \mathrm{d}x\\
     &=\int{\frac{x}{x^4(ax^2+b)}}\ \mathrm{d}x\\
     &=\int{\frac{1}{x^4(ax^2+b)}}\ \mathrm{d}\left(\frac{x^2}{2}\right) &&(x\ \mathrm{d}x=\mathrm{d}\left(\frac{x^2}{2}\right))\\
     &=\frac{1}{2}\int{\frac{1}{x^4(ax^2+b)}}\ \mathrm{d}(x^2)\\
     &=\frac{1}{2}\int{\left(\frac{A}{x^2}+\frac{B}{x^4}+\frac{C}{ax^2+b} \right)}\mathrm{d}(x^2) &&\text{(Partial Fraction Decomposition)}\\
     &=\frac{1}{2}\int{\frac{Ax^2(ax^2+b)+B(ax^2+b)+Cx^4}{x^4(ax^2+b)}}\ \mathrm{d}(x^2)\\ 
     &=\frac{1}{2}\int{\frac{(Aa+c)x^4+(Ab+Ba)x^2+Bb}{x^4(ax^2+b)}}\ \mathrm{d}(x^2)
\end{align*}
$$
Comparing coefficient gives: 
$$
(Aa+c)x^4+(Ab+Ba)x^2+Bb\equiv 1 \implies
\left\{
\begin{align*}
&Bb=1\\
&Ab+Ba=0\\
&Aa+C=0
\end{align*}
\right.
\implies
\left\{
\begin{align*}
&A=-\frac{a}{b^2}\\
&B=\frac{1}{b}\\
&C=\frac{a^2}{b^2}
\end{align*}
\right.
$$
Plug in back: 
$$
\begin{align*}
    I&=\frac{1}{2}\int{\left(\frac{-\frac{a}{b^2}}{x^2}+\frac{\frac{1}{b}}{x^4}+\frac{\frac{a^2}{b^2}}{ax^2+b} \right)}\mathrm{d}(x^2)\\
     &=\frac{1}{2}\left[-\frac{a}{b^2}\int{\frac{1}{x^2}}\ \mathrm{d}(x^2)+\frac{1}{b}\int{\frac{1}{(x^2)^2}}\ \mathrm{d}(x^2)+\frac{a^2}{b^2}\int{\frac{1}{ax^2+b}}\ 
     \frac{\mathrm{d}(ax^2+b)}{a} \right] &&\text{(Direct Substitution)}\\
     &=\frac{1}{2}\left(-\frac{a}{b^2}\ln{\left|x^2 \right|}-\frac{1}{bx^2}+\frac{a}{b^2}\ln{\left|ax^2+b \right|} +C\right)\\
     &=\frac{a}{2b^2}\ln{\frac{\left|ax^2+b\right|}{x^2}}-\frac{1}{2bx^2}+C \tag*{$\blacksquare$}
\end{align*}
$$

### 28
$$
\int \frac{1}{\left(ax^2+b\right)^2}\ \mathrm{d}x=\frac{x}{2b\left(ax^2+b\right)}+\frac{1}{2b}\int \frac{1}{ax^2+b}\ \mathrm{d}x
$$
<br/>
<details>
<summary><span className="math-keyword">Derivation</span></summary>

$$
\begin{align*}
    I&=\int{\frac{1}{(ax^2+b)^2}}\ \mathrm{d}x\\
     &=\int{\frac{1}{(ax^2+b)^2}}\ \frac{\mathrm{d}(ax^2+b)}{2ax} &&\text{(Direct Substitution)}\\
     &=\int{\frac{1}{2ax}}\ \left(-\frac{1}{ax^2+b} \right)\\
     &=-\frac{1}{2ax(ax^2+b)}-\int{-\frac{1}{ax^2+b}}\ \mathrm{d}\left(\frac{1}{2ax} \right) &&\text{(Integral by Parts)}\\
     &=-\frac{1}{2ax(ax^2+b)}-\frac{1}{2a}\int{\frac{1}{x^2(ax^2+b)}}\ \mathrm{d}x &&(\mathrm{d}\left(\frac{1}{2ax}\right)=-\frac{1}{2ax^2}\ \mathrm{d}x)\\
     &=-\frac{1}{2ax(ax^2+b)}-\frac{1}{2a}\left[-\frac{1}{bx}-\frac{a}{b}\int \frac{1}{ax^2+b}\ \mathrm{d}x\right] &&(\text{Integral Formula \#26})\\
     &=-\frac{1}{2ax(ax^2+b)}-\left(-\frac{1}{2abx}-\frac{1}{2b}\int{\frac{1}{x^2(ax^2+b)}}\ \mathrm{d}x  \right)\\
     &=-\frac{b}{2abx(ax^2+b)}+\frac{ax^2+b}{2abx(ax^2+b)}+\frac{1}{2b}\int{\frac{1}{x^2(ax^2+b)}}\ \mathrm{d}x\\
     &=\frac{ax^2}{2abx(ax^2+b)}+\frac{1}{2b}\int{\frac{1}{x^2(ax^2+b)}}\ \mathrm{d}x\\
     &=\frac{x}{2b\left(ax^2+b\right)}+\frac{1}{2b}\int \frac{1}{ax^2+b}\ \mathrm{d}x \tag*{$\blacksquare$}
\end{align*}
$$
